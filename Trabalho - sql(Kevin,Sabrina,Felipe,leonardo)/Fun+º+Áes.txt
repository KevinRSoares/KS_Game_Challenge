---Função de calculo de media de pontuação 
delimiter $
CREATE FUNCTION media_potuacao_j(CoJog int) 
RETURNS int
BEGIN 
	Declare MedPont int default 0;
    Select Avg(PonPar) from Partida Where not isnull(DatHorFimPar) and CodUsu = CoJog into MedPont;
	
RETURN MedPont;
END
$
---Função de calculo de media de pontuação

Obs.: O calculo é feito baseado nas partidas cujo campo DatHorFimPar seja diferente de nulo;
